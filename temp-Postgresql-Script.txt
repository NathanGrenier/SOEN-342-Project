CREATE TABLE Instructor (
    instructorID SERIAL PRIMARY KEY,
    iname VARCHAR(25),
    phone VARCHAR(15),
    availability VARCHAR(25),
    specialization VARCHAR(25)
);

CREATE TABLE Lesson (
    lessonID SERIAL PRIMARY KEY,
    lename VARCHAR(25),
    ltype VARCHAR(25),
    isGroupLesson BOOLEAN
);

CREATE TABLE Location (
    locationID SERIAL PRIMARY KEY,
    loname VARCHAR(25)
);

CREATE TABLE TimeSlot (
    timeslotID SERIAL PRIMARY KEY,
    startTime TIME,
    endTime TIME,
    dayOfWeek VARCHAR(10)
);

CREATE TABLE Schedule (
    scheduleID SERIAL PRIMARY KEY,
    month VARCHAR(10),
    year_ INTEGER
);

CREATE TABLE SequenceOf (
    relationID SERIAL PRIMARY KEY,
    scheduleID INTEGER,
    timeslotID INTEGER,
    FOREIGN KEY (scheduleID) REFERENCES Schedule(scheduleID),
    FOREIGN KEY (timeslotID) REFERENCES TimeSlot(timeslotID)
);

CREATE TABLE Offer (
    offeringID SERIAL PRIMARY KEY,
    lessonID INTEGER,
    locationID INTEGER,
    scheduleID INTEGER,
    instructorID INTEGER,
    status VARCHAR(10),
    FOREIGN KEY (lessonID) REFERENCES Lesson(lessonID),
    FOREIGN KEY (locationID) REFERENCES Location(locationID),
    FOREIGN KEY (scheduleID) REFERENCES Schedule(scheduleID),
    FOREIGN KEY (instructorID) REFERENCES Instructor(instructorID)
);

CREATE TABLE Client (
    clientID SERIAL PRIMARY KEY,
    cname VARCHAR(25)
);

CREATE TABLE OfferedTo (
    relationID SERIAL PRIMARY KEY,
    clientID INTEGER,
    offeringID INTEGER,
    FOREIGN KEY (clientID) REFERENCES Client(clientID),
    FOREIGN KEY (offeringID) REFERENCES Offer(offeringID)
);
